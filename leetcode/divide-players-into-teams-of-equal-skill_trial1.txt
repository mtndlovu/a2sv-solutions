from math import prod

class Solution:
    def dividePlayers(self, skill: List[int]) -> int:
        skill.sort( )
        n = len(skill)
        left, right = 0, n - 1
        target_sum = skill[left] + skill[right]
        total_chemistry = 0

        while left < right:
            if skill[left] + skill[right] != target_sum:
                return -1
                
            total_chemistry += skill[left] * skill[right]
            left += 1
            right -= 1
            
        return total_chemistry
