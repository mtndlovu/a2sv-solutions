class Solution:
    def maximumSubarraySum(self, nums: List[int], k: int) -> int:
        seen = set()
        left = 0
        total = 0
        result = 0

        for right, num in enumerate(nums):
            while num in seen:
                seen.remove(nums[left])
                total -= nums[left]
                left += 1

            seen.add(num)
            total += num

            if right - left + 1 == k:
                result = max(result, total)
                seen.remove(nums[left])
                total -= nums[left]
                left += 1

        return result
