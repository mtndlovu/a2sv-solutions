class Solution:
    def countPairs(self, nums: List[int], target: int) -> int:
        nums.sort()
        count = 0
        left, right = 0, len(nums) - 1
        
        while left < right:
            while nums[left] + nums[right] >= target:
                right -= 1
                if right == left:
                    break
            count += right - left
            left += 1
        return count
