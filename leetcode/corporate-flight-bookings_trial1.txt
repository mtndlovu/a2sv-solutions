class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        diff = [0] * (n+1)
        
        for start, end, seats in bookings:
            diff[start-1] += seats
            if end < n:
                diff[end] -= seats
        
        for i in range(1, n):
            diff[i] += diff[i-1]
        diff.pop()
        return diff
